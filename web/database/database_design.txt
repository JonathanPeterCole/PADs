Database Name: pads_db

tbl_cabinets:
  (PK) id - STRING
  location - STRING
  postcode - STRING

tbl_status:
  (PK) id - AUTO_INCREMENT INT
  (FK) cabinet_id - STRING
  door_status - STRING ("Open" / "Closed")
  hanger_status - STRING ("Down" / "Up")
  alarm_status - STRING ("Working" / "Fault")
  temp_status - INT (Current sensor value)
  last_update - CURRENT TIMESTAMP

tbl_guardians:
  (PK) id - AUTO_INCREMENT INT
  first_name - STRING
  last_name - STRING
  phone_number - STRING
  (FK) cabinet_id - STRING

  tbl_fleet:
    (PK) id - AUTO_INCREMENT INT
    first_name - STRING
    last_name - STRING
    phone_number - STRING
