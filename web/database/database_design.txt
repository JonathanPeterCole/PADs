Database Name: pads_db

tbl_cabinets:
  (PK) id - STRING
  location - STRING
  postcode - STRING

tbl_status:
  (PK) id - AUTO_INCREMENT INT
  (FK) cabinet_id - STRING
  door_status - TINYINT (0 = Closed / 1 = Open)
  hanger_status - TINYINT (0 = Up / 1 = Down)
  alarm_status - TINYINT (0 = Working / 1 = Fault)
  temp_status - INT (Current sensor value)
  last_update - CURRENT TIMESTAMP

tbl_guardians:
  (PK) id - AUTO_INCREMENT INT
  first_name - STRING
  last_name - STRING
  phone_number - STRING
  (FK) cabinet_id - STRING

  tbl_fleet:
    (PK) id - AUTO_INCREMENT INT
    first_name - STRING
    last_name - STRING
    phone_number - STRING
